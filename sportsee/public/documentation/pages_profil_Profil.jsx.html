/** 
 * @namespace Profil
 * @function Profil
 * @description this component renders a user profile page with various charts and information cards
* organizes structure around graphic elements.
* @returns {JSX.Element}.
*/
function Profil() {
    const { datas } = useContext(datasContext);
    const levelSentence = "Félicitation ! Vous avez explosé vos objectifs hier 👏";
    const userName = { firstName: datas?.userDatas.userInfos.firstName, lastName: datas?.userDatas.userInfos.lastName }
    if (!datas) return null;
    return <MainProfil>
        <UserCard>
            <UserName>
                <Bienvenue>Bonjour <span>{(userName.firstName)}
                </span></Bienvenue>
                <Sentence>{levelSentence}</Sentence>
            </UserName>
        </UserCard>
        <Container>
            <MainContainerGraphic>
                <BarChart />
                <MainGroup>
                    <Linechart />
                    <Radarchart />
                    <Radialchart />
                </MainGroup>
            </MainContainerGraphic>
            <CardsInfos />
        </Container>

    </MainProfil>;
}

export default Profil;
